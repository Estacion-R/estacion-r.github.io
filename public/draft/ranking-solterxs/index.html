tabla_graf <- centroides_aglomerados %>% 
  mutate(AGLOMERADO = as.factor(AGLOMERADO)) %>%

  mutate(
    col_y = rank(-porc) %>%

    sigmoid_end = 22,
    col_x_start = 25,
    col_x_end = porc %>%  
      data_rescale(to=c(25, 110), range = c(0, max(porc))),

  )
